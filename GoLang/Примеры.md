
```Go
package main

import "fmt"

func main() {
	var n int
	fmt.Scan(&n)
	var hour, minute int
	hour = n / 60
	minute = n % 60
	fmt.Printf("%d мин - это %d час %d минут.", n, hour, minute)
}
```
